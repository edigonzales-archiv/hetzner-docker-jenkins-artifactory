#cloud-config
users:
  - name: appuser
    shell: /bin/bash

package_upgrade: true

runcmd:
  - apt update
  - apt install apt-transport-https ca-certificates curl software-properties-common
  - curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
  - add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu bionic stable"
  - apt update
  - apt-cache policy docker-ce
  - apt install -y docker-ce
  - curl -L https://github.com/docker/compose/releases/download/1.23.2/docker-compose-`uname -s`-`uname -m` -o /usr/local/bin/docker-compose
  - chmod +x /usr/local/bin/docker-compose
  - usermod -aG docker appuser
  - mkdir -p /var/jenkins_home
  - chown -R appuser:appuser /var/jenkins_home
  - git clone https://github.com/edigonzales/hetzner-docker-jenkins-artifactory.git /home/appuser/hetzner-docker-jenkins-artifactory
  - chown -R appuser:appuser /home/appuser/hetzner-docker-jenkins-artifactory
  - sudo -u appuser docker-compose -f /home/appuser/hetzner-docker-jenkins-artifactory/modules/docker-compose.yml build
  - sudo -u appuser docker-compose -f /home/appuser/hetzner-docker-jenkins-artifactory/modules/docker-compose.yml up

power_state:
  delay: "now"
  mode: reboot
  message: First reboot
  condition: True
